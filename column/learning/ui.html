		label: 'title',
		emitPath: false, //emitPath为false时，v-model的绑定值为选中的当前节点id,类型为字符串;emitPath为true时,v-model的绑定值为根节点到选中的当前节点id组成的数组。(默认为true)

// 注意:一般列表页搜索时不设置emitPath或者设置emitPath:true,回显的时候设置emitPath:false

vxe-table 中,遍历表格数组，对每一行的对象数据调用组件封装的方法时尽量使用 for 循环而不是 forEach，因为 forEach 遍历出来的 item 是一个全新的对象，会导致调用方法失败。如果一定要使用 forEach，应该使用 index 来获取每一项而不是用 item。代码示例: ​

js
// 正确代码
nextTick(() => {
  for (let i = 0; i < state.selectedData.length; i++) {
    console.log(refTable.value, "refTable.value++++");
    refTable.value.setCheckboxRow(state.selectedData[i], true);
  }
});
// 正确代码
nextTick(() => {
  if (state.selectedData.length) {
    state.selectedData.forEach((item, index) => {
      refTable.value.setCheckboxRow(state.selectedData[index], true);
    });
  }
});
// 错误代码
nextTick(() => {
  if (state.selectedData.length) {
    state.selectedData.forEach((item, index) => {
      refTable.value.setCheckboxRow(item, true);
    });
  }
});

elementPlus 的 el-table 组件中 show-overflow-tooltip 属性 ​

该属性在以下几种情况下会失效: 1.插槽中手动设置了 css 样式的溢出显示省略号 2.插槽根元素用了 div 标签 3.插槽根元素设置了 display: flex 解决方法:手动设置 css 样式的溢出显示省略号，再手动添加 tooltip 组件

js
<el-table>
	  <el-table-column prop="productName" label="产品名称">
			<template #default="{ row }">
				<el-tooltip placement="top" :visible="row.visible">
					<template #content>
						{{ row.productName }}
					</template>
					<span
						class="productName"
						@mouseenter="row.visible = true"
						@mouseleave="row.visible = false"
					>
						<span class="span-txt">{{ row.productName }}</span>
						<el-icon v-if="row.pictureUrls" color="#1890FF" size="18px">
							&nbsp;
							<PictureFilled />
						</el-icon>
					</span>
				</el-tooltip>
			</template>
		</el-table-column>
	</el-table>
.productName {
	display: flex;
	align-items: center;
	.span-txt {
		overflow: hidden;
		text-overflow: ellipsis;
		white-space: nowrap;
	}
}

表格里的 el-select 的数据 option 如果是每次选择后动态过滤已选择的数据导致数据回显异常 ​

因为过滤已选中数据的原因，回显的时候 value 对应的值实际上已经从 option 里面过滤掉了,无法根据 value 回显对应的 label，就会直接显示成 value 处理思路:直接存两份数据，一份保存过滤前的原始数据 originOption,一份保存过滤后的数据 filterOption;每次下拉框隐藏的时候将 option 赋值为 filterOption，每次回显的时候讲 option 赋值为 originOption

修改 elementUI 中轮播图的动画效果 ​

自定义一个轮播动画覆盖默认动画: 常用值:linear(线性即匀速),ease(慢到快到慢,相对于匀速:中间快，两头慢),ease-in(慢进快出),ease-out(快进慢出),ease-in-out(慢到快到慢,相对于匀速:两头慢)

js
/* 自定义轮播 */

  transition: transform 1s ease-out;
}

elementUI 中 el-dropdown 组件中的@command 方法优化 ​

因为子组件 el-dropdown-menu 的 command 属性值可以为字符串、数值、对象,可以将 command 属性设为一个对象,将函数作为该对象的方法传入,然后在父组件 el-dropdown 组件中的@command 方法里面直接调用传入对象的方法,即可实现子方法的映射，省去了 if 或者 switch 的这种条件判断

html
<el-dropdown @command="(command) => handleOperationFun(command, row)">
  <el-button link type="primary">操作</el-button>
  <template #dropdown>
    <el-dropdown-menu>
      <el-dropdown-item :command="{ command: handleWithdraw }">
        <el-button link type="default">
          <el-icon><Document /></el-icon>
          撤回
        </el-button>
      </el-dropdown-item>
      <el-dropdown-item :command="{ command: handleCancel }">
        <el-button link type="default">
          <el-icon><Document /></el-icon>
          取消
        </el-button>
      </el-dropdown-item>
      <el-dropdown-item :command="{ command: handleDelete }">
        <el-button link type="default">
          <el-icon><Document /></el-icon>
          删除
        </el-button>
      </el-dropdown-item>
    </el-dropdown-menu>
  </template>
</el-dropdown>
js
// 操作按钮
const handleOperationFun = (command: any, row: any) => {
  console.log(command, row, "操作");
  command.command(row);
};
// 撤回按钮
const handleWithdraw = () => {
  console.log("撤回");
};

// 取消按钮
const handleCancel = () => {
  console.log("取消");
};

// 删除按钮
const handleDelete = (row?: any) => {
  console.log("删除", row);
};

elementUI 中 el-date-picker 组件禁用指定日期之前或之后的日期 ​

需要使用 disabled-date 属性绑定对应的函数进行处理

html
<el-date-picker
  v-model="row.planEndTime"
  clearable
  :disabled-date="pickerOption"
  format="YYYY-MM-DD HH:mm"
  placeholder="----年--月--日 --时--分"
  type="datetime"
  value-format="YYYY-MM-DD HH:mm:ss"
/>
js
const pickerOption = (date: any) => {
  if (state.formData.planTimeArr?.length) {
    const start = new Date(state.formData.planTimeArr[0]);
    const end = new Date(state.formData.planTimeArr[1]);
    return date.getTime() < start.getTime() || date.getTime() > end.getTime();
  }
};

elementUI 中 el-select 组件给change函数传递每项数据的多个属性值的写法 ​

通过value-key="id"定义一个key,将value绑定一个对象,对象的属性可以自定义,属性值绑定为每项数据的各个属性即可

html
<el-select
  v-model="row.processIdObj"
  clearable
  value-key="id"
  @change="(value) => changeProcessId(value, row)"
>
  <el-option
    v-for="item in processOption"
    :key="item.processId"
    :label="item.processLineNumberName"
    :value="{  
			id: item.processId,
      number: item.processLineNumber,
      detailId: item.processDetailId,
    }"
  />
</el-select>
